## Step 5: Update image generation to use contentType for model routing in `server/adminRoutes.ts`

Update POST /api/admin/marketing/queue/:id/generate-image to auto-select model based on contentType:

const item = await storage.getMarketingQueueItem(id);
const contentType = item.contentType || "feed_image";

// Auto-select model based on content type if not explicitly provided
let selectedModel = model;
if (!model || model === "auto") {
  if (contentType === "card_news") {
    selectedModel = "ideogram";
  } else if (contentType === "reel" || contentType === "tiktok") {
    selectedModel = "kling";
  } else {
    selectedModel = "flux-pro";
  }
}
