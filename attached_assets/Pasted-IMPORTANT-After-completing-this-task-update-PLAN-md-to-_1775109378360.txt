Replace generateImage() with:

export async function generateImage({
  prompt,
  model = "nano-banana-2",
  referenceImageUrls = [],
  aspectRatio = "1:1",
}: {
  prompt: string;
  model?: ImageModel;
  referenceImageUrls?: string[];
  aspectRatio?: string;
}): Promise<{ imageUrl: string }> {

  // Nano Banana models — use edit variant if reference images provided
  if (model === "nano-banana-2" || model === "nano-banana-pro" || model === "nano-banana") {
    const endpointMap: Record<string, string> = {
      "nano-banana-2": referenceImageUrls.length > 0 ? "fal-ai/nano-banana-2/edit" : "fal-ai/nano-banana-2",
      "nano-banana-pro": referenceImageUrls.length > 0 ? "fal-ai/nano-banana-pro/edit" : "fal-ai/nano-banana-pro",
      "nano-banana": referenceImageUrls.length > 0 ? "fal-ai/nano-banana/edit" : "fal-ai/nano-banana",
    };
    const endpoint = endpointMap[model];

    const input: any = {
      prompt,
      aspect_ratio: aspectRatio,
      output_format: "png",
      safety_tolerance: "4",
      resolution: "1K",
    };

    if (referenceImageUrls.length > 0) {
      input.image_urls = referenceImageUrls.slice(0, 5);
    }

    const result = await fal.run(endpoint, { input }) as any;
    const imageUrl = result.images?.[0]?.url || "";
    if (!imageUrl) throw new Error(`${model} returned no image URL`);

    const { saveImageFromUrl } = await import("./storageService");
    const filename = `img_${Date.now()}_${Math.random().toString(36).slice(2)}`;
    const permanentUrl = await saveImageFromUrl(imageUrl, filename);
    return { imageUrl: permanentUrl };
  }

  // Ideogram V3 — card news
  if (model === "ideogram") {
    const result = await fal.run("fal-ai/ideogram/v3", {
      input: {
        prompt,
        aspect_ratio: aspectRatio === "9:16" ? "ASPECT_9_16" : "ASPECT_1_1",
        style: "PHOTO",
      },
    }) as any;
    const imageUrl = result.images?.[0]?.url || "";
    if (!imageUrl) throw new Error("Ideogram returned no image URL");
    const { saveImageFromUrl } = await import("./storageService");
    const filename = `img_${Date.now()}_${Math.random().toString(36).slice(2)}`;
    const permanentUrl = await saveImageFromUrl(imageUrl, filename);
    return { imageUrl: permanentUrl };
  }

  // Kling — video
  if (model === "kling") {
    const result = await fal.run("fal-ai/kling-video/v1.6/pro/text-to-video", {
      input: { prompt, duration: "5", aspect_ratio: "9:16" },
    }) as any;
    const videoUrl = result.video?.url || "";
    if (!videoUrl) throw new Error("Kling returned no video URL");
    const { saveVideoFromUrl } = await import("./storageService");
    const filename = `video_${Date.now()}_${Math.random().toString(36).slice(2)}`;
    const permanentUrl = await saveVideoFromUrl(videoUrl, filename);
    return { imageUrl: permanentUrl };
  }

  throw new Error(`Unknown model: ${model}`);
}

## Task 2: Update server/adminRoutes.ts — generate-image endpoint

Accept model override from request body. Auto-select based on contentType but allow manual override:

app.post("/api/admin/marketing/queue/:id/generate-image", requireAdmin, async (req, res) => {
  try {
    const { id } = req.params;
    const { model: modelOverride } = req.body;

    const item = await storage.getMarketingQueueItem(id);
    if (!item) return res.status(404).json({ error: "Queue item not found" });

    const prompt = item.imagePrompt || item.caption || "";
    if (!prompt) return res.status(400).json({ error: "No image prompt available" });

    // Get Gukdung reference images
    const gukdungImages = await storage.getGukdungImages();
    const referenceImages = gukdungImages
      .filter(img => img.isTrainingData && img.imageUrl)
      .slice(0, 5)
      .map(img => img.imageUrl);

    // Auto-select model and aspect ratio from contentType
    const contentType = item.contentType || "feed_image";
    let autoModel: string = "nano-banana-2";
    let aspectRatio = "1:1";

    if (contentType === "card_news") {
      autoModel = "ideogram";
    } else if (contentType === "reel" || contentType === "tiktok") {
      autoModel = "kling";
      aspectRatio = "9:16";
    } else if (contentType === "story_image") {
      autoModel = "nano-banana-2";
      aspectRatio = "9:16";
    }

    // Use manual override if provided
    const selectedModel = modelOverride || autoModel;

    const { generateImage } = await import("./services/falService");
    const result = await generateImage({
      prompt: `${prompt}. Professional pet photography style, high quality.`,
      model: selectedModel as any,
      referenceImageUrls: selectedModel !== "ideogram" && selectedModel !== "kling" ? referenceImages : [],
      aspectRatio,
    });

    if (!result.imageUrl) {
      return res.status(500).json({ error: "Image generation failed" });
    }

    const updated = await storage.updateMarketingQueueItem(id, { imageUrl: result.imageUrl });
    res.json({ success: true, imageUrl: result.imageUrl, item: updated });
  } catch (error: any) {
    console.error("FAL.AI generate image error:", error);
    res.status(500).json({ error: error.message || "Image generation failed" });
  }
});
